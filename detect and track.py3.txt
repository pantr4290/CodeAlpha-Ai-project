from ultralytics import YOLO


def main():
    """
    STEP 1: Set up real-time video input using webcam.
    STEP 2: Use pre-trained YOLO model for object detection.
    STEP 3: Apply object tracking (DeepSORT/BOTSORT via Ultralytics).
    STEP 4: Display video with bounding boxes, labels, and tracking IDs.
    """
    # Load a pre-trained YOLOv8 model (automatically downloads if not present)
    model = YOLO("yolov8n.pt")  # nano model for speed

    # Run tracking on webcam (source=0)
    # 'show=True' opens a window with bounding boxes and tracking IDs.
    # 'tracker' uses a config file (botsort.yaml by default).
    model.track(
        source=0,         # 0 = default webcam
        show=True,        # show output in a window
        stream=False,     # process as a continuous stream
        tracker="botsort.yaml",  # tracking algorithm
        verbose=True
    )


if __name__ == "__main__":
    main()
